        if (!DrawingPool.Instance.IsInitialized || DrawingPool.Instance.CurrentSeed != seed)
        {
            DrawingPool.Instance.Initialize(seed);
        }

        InitShaders();
        _wallMat = MakeMaterial(DrawingPool.Instance.GetWall(0), cellSize / 2f, wallHeight / 2f);
        _floorMat = MakeMaterial(DrawingPool.Instance.GetFloor(0), 1f, 1f);
        _ceilMat = MakeMaterial(DrawingPool.Instance.GetCeiling(0), 1f, 1f);
        _handleMat = MakeMaterial(DrawingPool.Instance.GetTorch(0), 1f, 1f);
        
        var flameFrames = DrawingPool.Instance.GetFlameTextures();
        if (flameFrames != null && flameFrames.Length > 0)
            _flameMat = MakeTransparentMaterial(flameFrames[0]);
        else
            Debug.LogWarning("[MazeRenderer] No flame frames found!");
    }

    private static void InitShaders()
    {
        _litShader ??= Shader.Find("Universal Render Pipeline/Lit") ?? Shader.Find("URP/Lit") ?? Shader.Find("Standard");
        _unlitShader ??= Shader.Find("Universal Render Pipeline/Unlit") ?? Shader.Find("URP/Unlit") ?? Shader.Find("Unlit/Transparent");
        
        if (_unlitShader == null)
        {
            Debug.LogWarning("[MazeRenderer] Unlit shader not found! Trying Sprites/Default.");
            _unlitShader = Shader.Find("Sprites/Default");
        }
        if (_litShader == null)
        {
            Debug.LogWarning("[MazeRenderer] Lit shader not found! Trying Standard.");
            _litShader = Shader.Find("Standard");
        }
    }

    private Material MakeMaterial(Texture2D tex, float tilingX, float tilingY)
    {
        if (!_litShader)
        {
            Debug.LogError("[MazeRenderer] Lit shader is null! Cannot create wall material.");
            return null;
        }
        if (!tex)
        {
            Debug.LogError("[MazeRenderer] Wall texture is null!");
            return null;

    private void BuildGeometry()
    {
        var wallFaces = new System.Collections.Generic.List<(Vector3, Quaternion)>();
